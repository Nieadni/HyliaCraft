public VoxelShape makeShape(){
	VoxelShape shape = VoxelShapes.empty();
	shape = VoxelShapes.combineAndSimplify(shape, VoxelShapes.create(0.25, 0.03125, 0.25, 0.75, 0.53125, 0.75);
	shape = VoxelShapes.combineAndSimplify(shape, VoxelShapes.create(0.25, 0.03125, 0.25, 0.75, 0.53125, 0.75);

	return shape;
}
